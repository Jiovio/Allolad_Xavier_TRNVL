    Future<void> _fetchMoreMessages() async {
    setState(() {
      isLoading = true;
    });

    List<Messages> newMessages = await getMessages(limit, offset);
    
    setState(() {
      messages.insertAll(0, newMessages);
      offset += limit;
      isLoading = false;
    });
